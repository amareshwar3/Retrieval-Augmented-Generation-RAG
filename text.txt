"use client"

import * as React from "react"

import AppHeader from "../components/AppHeader"
import JobForm from "../components/JobForm"
import JobsTable from "../components/JobsTable"
import QueueTable from "../components/QueueTable"
import RecentLogs from "../components/RecentLogs"

import {
  JobFormState,
  JobOut,
  QueueOut,
} from "../lib/types"

import {
  toJobCreatePayload,
} from "../lib/mappers"

import {
  apiDelete,
  apiGet,
  apiPatch,
  apiPost,
} from "../lib/api"

const initialState: JobFormState = {
  version: "local",
  debugMode: "True",
  source: "offline",

  filePathDataRepMapping:
    "DataRepresentativenessMapping.xlsx",

  filePathMvp2Output: "glcs_data.csv",

  filePathIdqData: "IDQ_DATA.xlsx",

  filePathImrData: "IMR_Ratings.xlsx",

  filePathRagStaticData:
    "RAG_STATIC_DATA.xlsx",

  submittedBy: "",

  cronExpression: "",

  timezone: "UTC",
}

export default function Page() {
  const [state, setState] =
    React.useState<JobFormState>(
      initialState
    )

  const [submitStatus, setSubmitStatus] =
    React.useState<string>("")

  const [jobs, setJobs] =
    React.useState<JobOut[]>([])

  const [queue, setQueue] =
    React.useState<QueueOut[]>([])

  const [jobsLoading, setJobsLoading] =
    React.useState(true)

  const [queueLoading, setQueueLoading] =
    React.useState(true)

  const [jobsError, setJobsError] =
    React.useState("")

  const [queueError, setQueueError] =
    React.useState("")

  const loadJobs = React.useCallback(
    async () => {
      setJobsLoading(true)
      setJobsError("")

      try {
        const data =
          await apiGet<JobOut[]>(
            "/jobs?limit=50&offset=0"
          )

        setJobs(
          Array.isArray(data)
            ? data
            : []
        )
      } catch (e: unknown) {
        const message =
          e instanceof Error
            ? e.message
            : typeof e === "string"
            ? e
            : "Unknown error"

        setJobsError(message)
      } finally {
        setJobsLoading(false)
      }
    },
    []
  )

  const loadQueue =
    React.useCallback(
      async () => {
        setQueueLoading(true)
        setQueueError("")

        try {
          const data =
            await apiGet<QueueOut[]>(
              "/queue?limit=200&offset=0"
            )

          setQueue(
            Array.isArray(data)
              ? data
              : []
          )
        } catch (e: unknown) {
          const message =
            e instanceof Error
              ? e.message
              : typeof e === "string"
              ? e
              : "Unknown error"

          setQueueError(message)
        } finally {
          setQueueLoading(false)
        }
      },
      []
    )

  React.useEffect(() => {
    void loadJobs()
    void loadQueue()
  }, [loadJobs, loadQueue])

  const submit = async () => {
    try {
      const payload =
        toJobCreatePayload(state)

      const created =
        await apiPost<JobOut>(
          "/jobs",
          payload
        )

      setSubmitStatus(
        `Saved. request_id: ${
          created.request_id
        } (job_id: ${
          created.job_id ?? "-"
        })`
      )

      setState(initialState)

      await loadJobs()
      await loadQueue()
    } catch (e: unknown) {
      const message =
        e instanceof Error
          ? e.message
          : typeof e === "string"
          ? e
          : "Unknown error"

      setSubmitStatus(
        `Failed: ${message}`
      )
    }
  }

  const onDelete = async (
    requestId: number
  ) => {
    if (
      !confirm(
        `Delete request ${requestId}?`
      )
    ) {
      return
    }

    try {
      await apiDelete(
        `/jobs/${requestId}`
      )

      await loadJobs()
      await loadQueue()
    } catch (e: unknown) {
      const message =
        e instanceof Error
          ? e.message
          : typeof e === "string"
          ? e
          : "Unknown error"

      setJobsError(message)
    }
  }

  const onQueueUpdate = async (
    requestId: number,
    patch: {
      cron_expression?:
        | string
        | null
      timezone?:
        | string
        | null
      priority?: number
    }
  ) => {
    try {
      await apiPatch(
        `/queue/${requestId}`,
        patch
      )

      await loadQueue()
    } catch (e: unknown) {
      const message =
        e instanceof Error
          ? e.message
          : typeof e === "string"
          ? e
          : "Unknown error"

      setQueueError(message)
    }
  }

  const isRunning = (
    row: QueueOut
  ) => {
    const s = String(
      row.queue_status ?? ""
    ).toUpperCase()

    return (
      s === "RUNNING" ||
      s === "SUBMITTED" ||
      s === "IN_PROGRESS"
    )
  }

  return (
    <div className="min-h-screen">
      <AppHeader title="RAG Job Scheduler" />

      <main className="mx-auto w-full max-w-6xl px-4 py-6">
        <div className="space-y-6">
          <JobForm
            state={state}
            setState={setState}
            onSubmit={submit}
            submitStatus={
              submitStatus
            }
            onReset={() => {
              setState(
                initialState
              )
              setSubmitStatus("")
            }}
          />

          <JobsTable
            rows={jobs}
            loading={jobsLoading}
            error={jobsError}
            onRefresh={loadJobs}
            onDelete={onDelete}
          />

          <QueueTable
            rows={queue}
            loading={
              queueLoading
            }
            error={queueError}
            onRefresh={
              loadQueue
            }
            onUpdate={
              onQueueUpdate
            }
            isRunning={
              isRunning
            }
          />

          <RecentLogs
            jobs={jobs}
            loading={
              jobsLoading
            }
            error={jobsError}
            onRefresh={
              loadJobs
            }
          />
        </div>
      </main>
    </div>
  )
}
