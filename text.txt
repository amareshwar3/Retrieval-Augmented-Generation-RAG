// frontend/components/JobsTable.tsx
"use client"

import * as React from "react"
import Link from "next/link"

import { JobOut } from "@/lib/types"
import SectionCard from "./SectionCard"

import {
  Table,
  TableBody,
  TableCell,
  TableHead,
  TableHeader,
  TableRow,
} from "@/components/ui/table"

import { Button } from "@/components/ui/button"

function clip(s?: string | null, max = 28) {
  if (!s) return "—"
  return s.length > max ? `${s.slice(0, max - 1)}…` : s
}

export default function JobsTable({
  rows,
  loading,
  error,
  onRefresh,
  onDelete,
}: {
  rows: JobOut[]
  loading: boolean
  error: string
  onRefresh: () => Promise<void>
  onDelete: (requestId: number) => Promise<void>
}) {
  return (
    <SectionCard title="REQUESTS / JOBS">
      <div className="space-y-4">
        {/* actions */}
        <div className="flex flex-wrap items-center gap-2">
          <Button
            variant="outline"
            onClick={() => void onRefresh()}
            disabled={loading}
          >
            Refresh
          </Button>
        </div>

        {/* messages */}
        {error ? (
          <p className="text-sm text-destructive">{error}</p>
        ) : null}

        {loading ? (
          <p className="text-sm text-muted-foreground">
            Loading...
          </p>
        ) : null}

        {/* table */}
        {!loading ? (
          <div className="overflow-auto rounded-xl border">
            <Table>
              <TableHeader>
                <TableRow>
                  <TableHead className="whitespace-nowrap">
                    Request ID
                  </TableHead>
                  <TableHead className="whitespace-nowrap">
                    Job ID
                  </TableHead>
                  <TableHead className="whitespace-nowrap">
                    Name
                  </TableHead>
                  <TableHead className="whitespace-nowrap">
                    Submitted By
                  </TableHead>
                  <TableHead className="whitespace-nowrap">
                    Status
                  </TableHead>
                  <TableHead className="whitespace-nowrap">
                    View
                  </TableHead>
                  <TableHead className="whitespace-nowrap">
                    Delete
                  </TableHead>
                </TableRow>
              </TableHeader>

              <TableBody>
                {rows.map((r) => (
                  <TableRow
                    key={String(r.request_id)}
                  >
                    <TableCell className="whitespace-nowrap">
                      {r.request_id}
                    </TableCell>

                    <TableCell className="whitespace-nowrap">
                      {r.job_id ?? "—"}
                    </TableCell>

                    <TableCell
                      title={r.name}
                      className="max-w-[220px]"
                    >
                      {clip(r.name, 34)}
                    </TableCell>

                    <TableCell
                      title={r.submitted_by ?? ""}
                      className="max-w-[220px]"
                    >
                      {clip(r.submitted_by, 30)}
                    </TableCell>

                    <TableCell className="whitespace-nowrap">
                      {r.job_status ?? "—"}
                    </TableCell>

                    <TableCell className="whitespace-nowrap">
                      <Link
                        href={`/jobs/${r.request_id}`}
                        className="text-sm underline underline-offset-4"
                      >
                        View
                      </Link>
                    </TableCell>

                    <TableCell className="whitespace-nowrap">
                      <Button
                        size="sm"
                        variant="destructive"
                        onClick={() =>
                          void onDelete(r.request_id)
                        }
                      >
                        Delete
                      </Button>
                    </TableCell>
                  </TableRow>
                ))}

                {rows.length === 0 ? (
                  <TableRow>
                    <TableCell
                      colSpan={7}
                      className="text-sm text-muted-foreground"
                    >
                      No jobs found.
                    </TableCell>
                  </TableRow>
                ) : null}
              </TableBody>
            </Table>
          </div>
        ) : null}
      </div>
    </SectionCard>
  )
}
