"use client"

import * as React from "react"
import Link from "next/link"

import SectionCard from "./SectionCard"

import { JobOut } from "../lib/types"

import {
  Table,
  TableBody,
  TableCell,
  TableHead,
  TableHeader,
  TableRow,
} from "@/components/ui/table"

import { Button } from "@/components/ui/button"

function clip(s?: string, max = 28) {
  if (!s) return "—"
  return s.length > max
    ? `${s.slice(0, max - 1)}…`
    : s
}

export default function JobsTable({
  rows,
  loading,
  error,
  onRefresh,
  onDelete,
}: {
  rows: JobOut[]
  loading: boolean
  error: string
  onRefresh: () => Promise<void>
  onDelete: (
    requestId: number
  ) => Promise<void>
}) {
  return (
    <SectionCard title="REQUESTS / JOBS">
      <div className="space-y-4">
        {/* Header Actions */}
        <div className="flex flex-wrap items-center gap-3">
          <Button
            variant="outline"
            className="rounded-xl px-5"
            onClick={() => void onRefresh()}
            disabled={loading}
          >
            {loading ? "Refreshing..." : "Refresh"}
          </Button>

          {error ? (
            <p className="text-sm font-medium text-destructive">
              {error}
            </p>
          ) : null}

          {!error && loading ? (
            <p className="text-sm text-muted-foreground">
              Loading...
            </p>
          ) : null}
        </div>

        {/* Table */}
        {!loading ? (
          <div className="overflow-auto rounded-2xl border bg-card/70 backdrop-blur-sm">
            <Table>
              <TableHeader>
                <TableRow>
                  <TableHead className="whitespace-nowrap">
                    Request ID
                  </TableHead>

                  <TableHead className="whitespace-nowrap">
                    Job ID
                  </TableHead>

                  <TableHead className="whitespace-nowrap">
                    Version
                  </TableHead>

                  <TableHead className="whitespace-nowrap">
                    Debug
                  </TableHead>

                  <TableHead className="whitespace-nowrap">
                    Source
                  </TableHead>

                  <TableHead className="whitespace-nowrap">
                    Data Rep Mapping
                  </TableHead>

                  <TableHead className="whitespace-nowrap">
                    MVP2 Output
                  </TableHead>

                  <TableHead className="whitespace-nowrap">
                    IDQ Data
                  </TableHead>

                  <TableHead className="whitespace-nowrap">
                    IMR Data
                  </TableHead>

                  <TableHead className="whitespace-nowrap">
                    RAG Static
                  </TableHead>

                  <TableHead className="whitespace-nowrap">
                    Submitted By
                  </TableHead>

                  <TableHead className="whitespace-nowrap">
                    Status
                  </TableHead>

                  <TableHead className="whitespace-nowrap">
                    View
                  </TableHead>

                  <TableHead className="whitespace-nowrap">
                    Delete
                  </TableHead>
                </TableRow>
              </TableHeader>

              <TableBody>
                {rows.map((r) => (
                  <TableRow
                    key={String(r.request_id)}
                    className="hover:bg-muted/40"
                  >
                    <TableCell className="whitespace-nowrap font-medium">
                      {r.request_id}
                    </TableCell>

                    <TableCell className="whitespace-nowrap">
                      {r.job_id ?? "—"}
                    </TableCell>

                    <TableCell className="whitespace-nowrap">
                      {r.version}
                    </TableCell>

                    <TableCell className="whitespace-nowrap">
                      {r.debug_mode}
                    </TableCell>

                    <TableCell className="whitespace-nowrap">
                      {r.source}
                    </TableCell>

                    <TableCell
                      title={r.file_path_data_rep_mapping}
                    >
                      {clip(
                        r.file_path_data_rep_mapping
                      )}
                    </TableCell>

                    <TableCell
                      title={r.file_path_mvp2_output}
                    >
                      {clip(
                        r.file_path_mvp2_output
                      )}
                    </TableCell>

                    <TableCell
                      title={r.file_path_idq_data}
                    >
                      {clip(
                        r.file_path_idq_data
                      )}
                    </TableCell>

                    <TableCell
                      title={r.file_path_imr_data}
                    >
                      {clip(
                        r.file_path_imr_data
                      )}
                    </TableCell>

                    <TableCell
                      title={
                        r.file_path_rag_static_data
                      }
                    >
                      {clip(
                        r.file_path_rag_static_data
                      )}
                    </TableCell>

                    <TableCell className="whitespace-nowrap">
                      {r.submitted_by ?? "—"}
                    </TableCell>

                    <TableCell className="whitespace-nowrap">
                      {r.job_status ?? "—"}
                    </TableCell>

                    <TableCell className="whitespace-nowrap">
                      <Link
                        href={`/jobs/${r.request_id}`}
                        className="text-sm font-medium underline underline-offset-4 hover:text-primary"
                      >
                        View
                      </Link>
                    </TableCell>

                    <TableCell className="whitespace-nowrap">
                      <Button
                        variant="destructive"
                        size="sm"
                        className="rounded-lg"
                        onClick={() =>
                          void onDelete(
                            r.request_id
                          )
                        }
                      >
                        Delete
                      </Button>
                    </TableCell>
                  </TableRow>
                ))}

                {rows.length === 0 ? (
                  <TableRow>
                    <TableCell
                      colSpan={14}
                      className="py-8 text-center text-sm text-muted-foreground"
                    >
                      No jobs found.
                    </TableCell>
                  </TableRow>
                ) : null}
              </TableBody>
            </Table>
          </div>
        ) : null}
      </div>
    </SectionCard>
  )
}
