"use client"

import * as React from "react"
import Link from "next/link"

import SectionCard from "./SectionCard"
import { JobOut } from "../lib/types"

import {
  Table,
  TableBody,
  TableCell,
  TableHead,
  TableHeader,
  TableRow,
} from "@/components/ui/table"

import { Button } from "@/components/ui/button"

function clip(s: string, max = 42) {
  if (!s) return "—"
  return s.length > max ? s.slice(0, max - 1) + "…" : s
}

export default function RecentLogs({
  jobs,
  loading,
  error,
  onRefresh,
}: {
  jobs: JobOut[]
  loading: boolean
  error: string
  onRefresh: () => Promise<void>
}) {
  const items = jobs.slice(0, 10)

  const label = (j: JobOut) =>
    `${j.version} | ${j.debug_mode} | ${j.source} | ${clip(
      j.file_path_mvp2_output
    )}`

  return (
    <SectionCard title="RECENT JOB LOGS (ENTRY POINTS)">
      <div className="space-y-4">
        <Button
          variant="outline"
          onClick={() => void onRefresh()}
          disabled={loading}
        >
          Refresh
        </Button>

        {error ? (
          <p className="text-sm text-destructive">{error}</p>
        ) : null}

        {loading ? (
          <p className="text-sm text-muted-foreground">
            Loading…
          </p>
        ) : null}

        {!loading ? (
          <div className="overflow-auto rounded-md border">
            <Table>
              <TableHeader>
                <TableRow>
                  <TableHead className="whitespace-nowrap">
                    Request ID
                  </TableHead>

                  <TableHead>Job</TableHead>

                  <TableHead className="whitespace-nowrap">
                    Logs
                  </TableHead>
                </TableRow>
              </TableHeader>

              <TableBody>
                {items.map((j) => (
                  <TableRow
                    key={String(j.request_id)}
                  >
                    <TableCell className="whitespace-nowrap">
                      {j.request_id}
                    </TableCell>

                    <TableCell>
                      {label(j)}
                    </TableCell>

                    <TableCell className="whitespace-nowrap">
                      <Link
                        href={`/logs/${j.request_id}`}
                        className="text-sm underline underline-offset-4"
                      >
                        View logs
                      </Link>
                    </TableCell>
                  </TableRow>
                ))}

                {items.length === 0 ? (
                  <TableRow>
                    <TableCell
                      colSpan={3}
                      className="text-sm text-muted-foreground"
                    >
                      No jobs yet.
                    </TableCell>
                  </TableRow>
                ) : null}
              </TableBody>
            </Table>
          </div>
        ) : null}
      </div>
    </SectionCard>
  )
}
